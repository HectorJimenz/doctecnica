\begin{Arquitectura}{AG-GINF}{Arquitectura General del Proceso de Gestión de Infraestructura} {
		
		%-------------------------------------------
		%Descripción
		El proceso de Gestión de Infraestructura se encarga de 
		gestionar los diversos espacios del instituto para la impartición de las clases de las unidades de aprendizaje\footnote{ver \refElem{UnidadDeAprendizaje}}. Al mismo tiempo, se encarga de gestionar los espacios externos, que pertenecen a otras entidades, en las cuales el instituto tiene permitido impartir clases.
		
		%-------------------------------------------
		%Diagrama de arquitectura
		\noindent La Figura \cdtRefImg{arq:PP-RINF}{Arquitectura del Proceso de Registro de Infrestructura}
		 muestra la interacción con los procesos que le proporcionan la información necesaria para llevar a cabo sus funciones.
		
		\Pfig[.4]{pinf/imagenes/arquitectura-infraestructura.png}{arq:PP-RINF}{Arquitectura del Proceso de Registro de Infrestructura}

		%\Pfig[0.8]{RINF/imagenes/arquitectura-infraestructura.png}{arq:PRINF}{RINF Diagrama de Arquitectura del Proceso de Registro de Infrestructura}
		
	}{RINF}

\end{Arquitectura}

\begin{ADescripcion}
	\item \textbf{Gestión de Estructura Académica}. Recibe la información del proceso de Gestión de Infraestructura, en donde se escogerán los espacios en los cuáles se impartirán las Unidades de Aprendizaje para cada uno de los planes de estudio.

\end{ADescripcion}
