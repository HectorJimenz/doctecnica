\section{Validación de Cambios del Proceso de Gestión de Programas Académicos}
\label{section:PF-GPA:validacion}
%------------- Acuerdo 1 -------------
\hrule
\vspace{0.2cm}
\begin{Cdescription}
	\item[Subproceso:] Registrar \refElem{ProgramaAcademico}.
	\item[Situación actual:] La \refElem{DAE} es quien registra los Programas Académicos aprobados en el \refElem{SAES}, lo cual deriva en un incremento en el tiempo para tener los programas académicos actualizados con información oficial.
	\item[Perfil actual:] \refElem{DAE}.
	\item[Solución propuesta:] Las Unidades Académicas \footnote{ Ver \refElem{UnidadAcademica}} de Nivel Medio Superior o Nivel Superior, serán las encargadas de registrar o actualizar sus Programas Académicos en el \refElem{Calmecac} para ser validados por la \refElem{DES} o \refElem{DEMS} según corresponda.
	\item[Perfil propuesto:] \refElem{UnidadAcademica}.
	\item[Tipo:] Cambio.\\

\end{Cdescription}
